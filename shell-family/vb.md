```vb
Set xmlDoc = CreateObject("Microsoft.XMLDOM")  
xmlDoc.Async = "False"  
xmlDoc.Load("HARDWARE.xml")  
Set colNodes=xmlDoc.selectNodes("//HARDWARE/COMPUTER[@os='Windows XP']")  
For Each objNode in colNodes  
  Wscript.Echo objNode.Text   
Next
```
